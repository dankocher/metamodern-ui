## MetDatePicker

### Get started
Install [moment.js](https://momentjs.com/) before getting started.
If you need to change the language, import it from "moment/locale/".

### Import

```jsx
import { MetDatePicker } from "@metamodern.dev/metamodern-ui/";
import moment from "moment";
require("moment/locale/ru");
```

### Usage

```jsx
<MetDatePicker
    onChange={() => {
        console.log("It was changed!");
    }}
    style={style}
    className={className}
    dateFontClass={dateFontClass}
    calendarFontClass={calendarFontClass}
    selectMonthLabel={selectMonthLabel}
    selectYearLabel={selectYearLabel}
    defaultSelectedDate={defaultSelectedDate}
    type={type}
    dateFormat={dateFormat}
    language={language}
    customMoment={moment}
    calendarIcon={calendarIcon}
    arrowIcon={arrowIcon}
    mainColor={mainColor}
    bgColor={bgColor}
    mainHoverColor={mainHoverColor}
    headerColor={headerColor}
    headerHoverColor={headerHoverColor}
    weekDayFontColor={weekDayFontColor}
    primaryFontColor={primaryFontColor}
    hoverDateBgColor={hoverDateBgColor}
    selectedColor={selectedColor}
    selectedFontColor={selectedFontColor}
    secondaryFontColor={secondaryFontColor}
/>
```
